CREATE TABLE "trasportista" (
  "id_trasportista" serial PRIMARY KEY,
  "rut_empresa" varchar(12),
  "nombre_empresa" varchar(150)
);

CREATE TABLE "despacho" (
  "id_despacho" bigserial PRIMARY KEY,
  "id_pedido" bigint, /* REFERENCIA LÓGICA al MS Ventas */
  "id_trasportista" int,
  "codigo_seguimiento" varchar(100),
  "estado_despacho" varchar(50),
  "destinatario_nombre" varchar(200),
  "despacho_calle" varchar(150),
  "despacho_numero" varchar(20),
  "despacho_comuna" varchar(100),
  "despacho_region" varchar(100)
);

-- Clave foránea interna del MS Logística
ALTER TABLE "despacho" ADD FOREIGN KEY ("id_trasportista") REFERENCES "trasportista" ("id_trasportista") DEFERRABLE INITIALLY IMMEDIATE;
